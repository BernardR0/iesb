sudo apt-get update (atualiza biblioteca)
sudo apt-get install gcc (instala programa)
sudo dpkg -i {nome do arquivo .deb} ( instala programa)
mkdir {nome da pasta} (cria uma pasta)
cd {nome da pasta} (Acesssa a pasta)
cd .. (volta pro diretorio anterior)
ls (lista os arquivos da pasta)
ls -Fltr ( lista os arquivos em ordem de modificação)
mv {origem} {destino} ( serve para mover/renomear o arquivo)
code . ( abre o editor na pasta que está)
code {nome da pasta} (abre o editr na pasta informada)

gcc {nome do codigo fonte} (compila um código)
gcc {nome do codigo fonte} -o {nome do executavel} (compila um codigo 
e gera um executavel com nome especifico)

./{nome do executavel} (executa)
